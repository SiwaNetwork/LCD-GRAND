        # –î–µ–π—Å—Ç–≤–∏—è –∫–Ω–æ–ø–æ–∫

        self.lcd.draw_circle(self.player_x, self.player_y, 10, color=(0, 255, 0), fill=True)
        self.lcd.draw_text(f"Score: {self.score}", 10, 10, color=(255, 255, 255))
        self.lcd.update()

# –ó–∞–ø—É—Å–∫ –∏–≥—Ä—ã
